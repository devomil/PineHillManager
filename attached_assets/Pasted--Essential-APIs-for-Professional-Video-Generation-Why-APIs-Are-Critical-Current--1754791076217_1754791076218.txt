# Essential APIs for Professional Video Generation

## 🎯 **Why APIs Are Critical:**
- **Current Problem**: Agent is creating basic content from scratch
- **Solution**: Leverage professional APIs for rich content, animations, and effects
- **Result**: TV-commercial quality videos instead of basic slideshows

## 🤖 **Content Generation APIs**

### 1. **Hugging Face (FREE - 30k chars/month)**
```javascript
// Generate professional marketing copy
const generateMarketingScript = async (productData) => {
  const response = await fetch('https://api-inference.huggingface.co/models/microsoft/DialoGPT-large', {
    headers: { Authorization: `Bearer ${HF_TOKEN}` },
    method: 'POST',
    body: JSON.stringify({
      inputs: `Create a 30-second marketing script for ${productData.name}, a ${productData.category} that helps with ${productData.benefits.join(', ')}. Include problem statement, solution, benefits, and call to action.`
    })
  });
  
  return response.json();
};

// Generate benefit statements
const generateBenefits = async (productType) => {
  const response = await fetch('https://api-inference.huggingface.co/models/gpt2', {
    headers: { Authorization: `Bearer ${HF_TOKEN}` },
    method: 'POST',
    body: JSON.stringify({
      inputs: `Professional health benefits for ${productType}: 1.`,
      parameters: { max_length: 200, temperature: 0.7 }
    })
  });
};
```

### 2. **Alternative Free Text APIs**
```javascript
// Cohere (Free tier: 100 calls/month)
const coherent = require('cohere-ai');
coherent.init('YOUR_FREE_API_KEY');

const generateContent = async (prompt) => {
  const response = await coherent.generate({
    model: 'command',
    prompt: prompt,
    max_tokens: 300
  });
  return response.body.generations[0].text;
};

// Local AI with Ollama (Completely free)
const ollamaGenerate = async (prompt) => {
  const response = await fetch('http://localhost:11434/api/generate', {
    method: 'POST',
    body: JSON.stringify({
      model: 'llama2', // or mistral, codellama
      prompt: prompt,
      stream: false
    })
  });
  return response.json();
};
```

## 🎨 **Animation & Visual APIs**

### 1. **Lottie Animations (FREE)**
```javascript
// Professional animated icons and effects
import lottie from 'lottie-web';

class ProfessionalAnimations {
  async loadHealthAnimations() {
    const animations = {
      heartbeat: 'https://assets2.lottiefiles.com/packages/lf20_hTCwJG.json',
      pills: 'https://assets8.lottiefiles.com/packages/lf20_1pxqjqps.json',
      checkmarks: 'https://assets4.lottiefiles.com/packages/lf20_jbrw3hcz.json',
      molecules: 'https://assets1.lottiefiles.com/packages/lf20_8uonAC.json',
      loading: 'https://assets3.lottiefiles.com/packages/lf20_hK4EwN.json',
      success: 'https://assets9.lottiefiles.com/packages/lf20_xlkxtmul.json'
    };
    
    return animations;
  }
  
  renderLottieToCanvas(animationUrl, canvas, frame) {
    const animation = lottie.loadAnimation({
      container: canvas,
      renderer: 'canvas',
      loop: false,
      autoplay: false,
      path: animationUrl
    });
    
    animation.goToAndStop(frame, true);
    return canvas;
  }
}
```

### 2. **Remotion (React-based Videos)**
```javascript
// Professional video creation framework
import { Composition, useCurrentFrame, useVideoConfig } from 'remotion';

const ProductVideo = ({ productData }) => {
  const frame = useCurrentFrame();
  const { fps } = useVideoConfig();
  
  return (
    <div style={{ backgroundColor: '#f8fafc', width: '100%', height: '100%' }}>
      <Scene1 frame={frame} fps={fps} visible={frame < 180} />
      <Scene2 frame={frame} fps={fps} visible={frame >= 180 && frame < 360} />
      <Scene3 frame={frame} fps={fps} visible={frame >= 360} />
    </div>
  );
};

// Register composition
export const compositions = [
  {
    id: 'ProductVideo',
    component: ProductVideo,
    durationInFrames: 900, // 30 seconds at 30fps
    fps: 30,
    width: 1920,
    height: 1080
  }
];
```

## 🖼️ **Visual Asset APIs**

### 1. **Unsplash (FREE - 50 requests/hour)**
```javascript
// Professional stock imagery
const getStockImages = async (query) => {
  const response = await fetch(
    `https://api.unsplash.com/search/photos?query=${query}&per_page=10`,
    { headers: { Authorization: `Client-ID ${UNSPLASH_ACCESS_KEY}` } }
  );
  
  return response.json();
};

// Get health/medical themed backgrounds
const getBackgrounds = async () => {
  const medicalBgs = await getStockImages('medical office clean professional');
  const healthBgs = await getStockImages('health wellness blue gradient');
  return [...medicalBgs.results, ...healthBgs.results];
};
```

### 2. **Remove.bg (FREE - 50 images/month)**
```javascript
// Remove backgrounds from product images
const removeBackground = async (imageFile) => {
  const formData = new FormData();
  formData.append('image_file', imageFile);
  
  const response = await fetch('https://api.remove.bg/v1.0/removebg', {
    method: 'POST',
    headers: { 'X-Api-Key': REMOVE_BG_KEY },
    body: formData
  });
  
  return await response.blob();
};
```

## 🎵 **Audio APIs**

### 1. **ElevenLabs (FREE - 10k chars/month)**
```javascript
// Professional voiceover generation
const generateVoiceover = async (text) => {
  const response = await fetch('https://api.elevenlabs.io/v1/text-to-speech/21m00Tcm4TlvDq8ikWAM', {
    method: 'POST',
    headers: {
      'Accept': 'audio/mpeg',
      'Content-Type': 'application/json',
      'xi-api-key': ELEVENLABS_KEY
    },
    body: JSON.stringify({
      text: text,
      model_id: 'eleven_monolingual_v1',
      voice_settings: {
        stability: 0.5,
        similarity_boost: 0.5,
        style: 0.5,
        use_speaker_boost: true
      }
    })
  });
  
  return await response.blob();
};
```

## 🛠️ **Complete Integration Strategy**

### Phase 1: Content APIs
```javascript
class VideoContentGenerator {
  constructor() {
    this.huggingFace = new HuggingFaceAPI();
    this.ollama = new OllamaLocal();
  }
  
  async generateVideoContent(productData) {
    // Generate script
    const script = await this.huggingFace.generateScript(productData);
    
    // Generate benefits
    const benefits = await this.huggingFace.generateBenefits(productData.category);
    
    // Generate call-to-action
    const cta = await this.huggingFace.generateCTA(productData);
    
    return { script, benefits, cta };
  }
}
```

### Phase 2: Visual APIs
```javascript
class VideoVisualGenerator {
  constructor() {
    this.lottie = new LottieAnimations();
    this.unsplash = new UnsplashAPI();
    this.removeBg = new RemoveBgAPI();
  }
  
  async generateVisuals(productData) {
    // Get professional backgrounds
    const backgrounds = await this.unsplash.getBackgrounds('medical professional');
    
    // Remove product background
    const cleanProduct = await this.removeBg.process(productData.image);
    
    // Get animated icons
    const animations = await this.lottie.getHealthAnimations();
    
    return { backgrounds, cleanProduct, animations };
  }
}
```

### Phase 3: Audio APIs
```javascript
class VideoAudioGenerator {
  constructor() {
    this.elevenlabs = new ElevenLabsAPI();
    this.freesound = new FreesoundAPI();
  }
  
  async generateAudio(script) {
    // Professional voiceover
    const voiceover = await this.elevenlabs.generateVoice(script);
    
    // Background music
    const music = await this.freesound.getCorporateMusic();
    
    return { voiceover, music };
  }
}
```
